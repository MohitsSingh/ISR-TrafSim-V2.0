0046 global s c

0048 
0049 % Select the correct layer
0050 layer=find(round2(s.layer,0.001)==round2(time,0.001));
0051 
0052 if(isempty(layer))
0053     create_layer_isr_trafsim(round2(time,0.01));
0054     layer=size(s.layer,2);
0055 end
0056 
0057 % Rotation matrix
0058 m_rot=[cos(t) -sin(t) 0; sin(t) cos(t) 0; 0 0 1];
0059 
0060 % Calculate car relative points
0061 widht=c.car(n).width+s.layer_w;
0062 length=c.car(n).length+s.layer_l;
0063 cy=[-widht/2 widht/2 widht/2 -widht/2];
0064 cx=[length/2 length/2 -length/2 -length/2];
0065 
0066 % New points of robot
0067 p1t=m_rot*[cx(1);cy(1);1]; p2t=m_rot*[cx(2);cy(2);1]; p3t=m_rot*[cx(3);cy(3);1]; p4t=m_rot*[cx(4);cy(4);1];
0068 cornX=[p1t(1) p2t(1) p3t(1) p4t(1)];
0069 cornY=[p1t(2) p2t(2) p3t(2) p4t(2)];
0070 
0071 % Define polygon points
0072 xbox=[cornX cornX(1)]+x;
0073 ybox=[cornY cornY(1)]+y;
0074 
0075 % Merge the mask and the current occupation layer
0076 mask = poly2mask(xbox*s.occup_res2,ybox*s.occup_res2,size(s.ocmap2,1),size(s.ocmap2,2));
0077 s.ocmap2(:,:,layer)=s.ocmap2(:,:,layer)|mask;
