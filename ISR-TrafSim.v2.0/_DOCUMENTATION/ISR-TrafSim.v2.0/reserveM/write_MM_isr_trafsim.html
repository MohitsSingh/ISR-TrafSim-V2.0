0045 function [M,L] = write_MM_isr_trafsim(M,L,time,x,y,t,~,width,length,margin_w,margin_l,map_res,map_dimx,map_dimy)
0046 
0047 % Select the correct layer
0048 layer=find(L==round2(time,0.01));
0049 
0050 if(isempty(layer))
0051     [M,L]=create_L(M,L,round2(time,0.01),map_res,map_dimx,map_dimy);
0052     layer=length(L);
0053 end
0054 
0055 % Rotation matrix
0056 m_rot=[cos(t) -sin(t) 0; sin(t) cos(t) 0; 0 0 1];
0057 
0058 % Calculate car relative points
0059 widht=width+margin_w; length=length+margin_l;
0060 cy=[-widht/2 widht/2 widht/2 -widht/2];
0061 cx=[length/2 length/2 -length/2 -length/2];
0062 
0063 % New points of robot
0064 p1t=m_rot*[cx(1);cy(1);1]; p2t=m_rot*[cx(2);cy(2);1]; p3t=m_rot*[cx(3);cy(3);1]; p4t=m_rot*[cx(4);cy(4);1];
0065 cornX=[p1t(1) p2t(1) p3t(1) p4t(1)];
0066 cornY=[p1t(2) p2t(2) p3t(2) p4t(2)];
0067 
0068 % Define polygon points
0069 xbox=[cornX cornX(1)]+x;
0070 ybox=[cornY cornY(1)]+y;
0071 
0072 % Merge the mask and the current occupation layer
0073 mask = poly2mask(xbox*map_res,ybox*map_res,size(M,1),size(M,2));
0074 M(:,:,layer)=M(:,:,layer)|mask;
0075 
0076 end
