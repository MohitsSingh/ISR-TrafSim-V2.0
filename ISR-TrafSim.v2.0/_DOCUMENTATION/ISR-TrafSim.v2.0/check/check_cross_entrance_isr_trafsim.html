0045 function occ=check_cross_entrance_isr_trafsim(roadID)
0046 global s
0047 
0048 % Roads information
0049 px=s.ri(roadID,3);
0050 py=s.ri(roadID,4);
0051 t=s.ri(roadID,7);       % Road orientation
0052 [cornX, cornY] = pcorners_isr_trafsim(t,s.rw,s.rw);
0053 cornX=cornX+px; cornY=cornY+py;
0054 
0055 % Advance car corners for front in amp(2) meters
0056 cornX(1)=cornX(1)+s.amplitudeRoundControl(2)*cos(t); cornY(1)=cornY(1)+s.amplitudeRoundControl(2)*sin(t);
0057 cornX(2)=cornX(2)+s.amplitudeRoundControl(2)*cos(t); cornY(2)=cornY(2)+s.amplitudeRoundControl(2)*sin(t);
0058 
0059 % Back car corners for back in amp(1) meters
0060 cornX(3)=cornX(3)-s.amplitudeCrossControl(1)*cos(t); cornY(3)=cornY(3)-s.amplitudeCrossControl(1)*sin(t);
0061 cornX(4)=cornX(4)-s.amplitudeCrossControl(1)*cos(t); cornY(4)=cornY(4)-s.amplitudeCrossControl(1)*sin(t);
0062 
0063 % Compute box
0064 xbox=[cornX cornX(1)];
0065 ybox=[cornY cornY(1)];
0066 mask = poly2mask(xbox*s.occup_res,ybox*s.occup_res,size(s.ocmap,1),size(s.ocmap,2));
0067 out = s.ocmap(:,:).*uint8(mask);
0068 resMask=find(out~=0);
0069 occ=[];
0070 if(isempty(resMask) || length(resMask)==1)   % If not exist obstacles in selected space
0071     s.intersectCarCross(:,roadID)=Inf;
0072     if(s.cross_mode==1 || s.cross_mode==2)
0073         s.lights(2,roadID)=0;
0074     end
0075 else
0076     % Get IDs from occupation space
0077     carIDs=getCarIDFromMap_isr_trafsim(resMask);
0078     occ=[occ carIDs];
0079 end
0080 end
