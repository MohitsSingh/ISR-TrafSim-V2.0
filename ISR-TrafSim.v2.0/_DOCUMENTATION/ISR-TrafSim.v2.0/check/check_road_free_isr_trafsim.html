0066 c4y=py_+l*sin(t-pi/2);
0067 c3x=px_+l*cos(t+pi/2);
0068 c3y=py_+l*sin(t+pi/2);

0070 % Checks which the pixels that are within the polygon
0071 xbox=[c1x c2x c3x c4x c1x];
