0044 function ota_commu_run_isr_trafsim(max_time, log_file)
0045 global current_time edebug fdebug size_Event_list Event_list;
0046 
0047 % Event driven simulation loop
0048 size_Event_list=0;
0049 while 1
0050     [min_instant, min_index] = min([Event_list(:).instant]);
0051     if isempty(min_instant)
0052         break;
0053     end
0054     if min_instant > max_time
0055         break;
0056     end
0057     
0058     NewEvents = action(Event_list(min_index), log_file);      % Get new events from executing the latest 'action'
0059     current_time = min_instant;
0060     Event_list(min_index) = [];                     % Delete the latest event which has been just executed.
0061     if(size(NewEvents,1)>1)
0062         NewEvents=NewEvents';
0063     end
0064     Event_list = [NewEvents Event_list];   % Append new events generated by executing the latest event
0065     size_Event_list=size(Event_list);
0066     
0067     if fdebug
0068         if size_Event_list(2)
0069             disp('BEGIN ***************************')
0070             for i=1:size_Event_list(1)
0071                 if 1 %~strcmp(Event_list(i).type, 'backoff') && ~strcmp(Event_list(i).type, 'wait_for_channel') && ~strcmp(Event_list(i).type, 'backoff_start')
0072                     disp([num2str(Event_list(i).node,2) ' ' Event_list(i).type ' ' num2str(Event_list(i).instant)]);
0073                 end
0074             end
0075             disp('END ############################')
0076         end
0077         
0078     end
0079 end
